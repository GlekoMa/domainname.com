#写一个函数吧！
func <- function(n){
  z <- replicate(100,mean(sample(x,size = n)))
  data <- data.frame(
    x = 1:100,
    y = z
  )
  list(
    d_mean = mean(data$y),
    d_sd = sd(data$y)
  )
}
func(n = 300)
## $d_mean
## [1] 80.03777
## 
## $d_sd
## [1] 0.7293346
func(n = 1000)
## $d_mean
## [1] 79.98659

## [1] 0.3473152
