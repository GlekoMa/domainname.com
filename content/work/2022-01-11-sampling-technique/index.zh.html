#写一个函数吧！
func <- function(n){
  z <- replicate(100,mean(sample(x,size = n)))
  data <- data.frame(
    x = 1:100,
    y = z
  )
  list(
    d_mean = mean(data$y),
    d_sd = sd(data$y)
  )
}
func(n = 300)
## $d_mean
## [1] 79.97701
## 
## $d_sd
## [1] 0.6379348
func(n = 1000)
## $d_mean
## [1] 80.03828
## 
## $d_sd
## [1] 0.3578753

标准差是抽样误差的一种表现形式；在其他条件相同的情况下，样本量越大，抽样误差越小。
